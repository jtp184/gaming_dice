def initialize(initarg = nil, rul = :sum)
  if initarg.is_a?(Array)
    @dice = initarg
  elsif initarg.is_a?(Dice)
    @dice = Array(initarg)
  end

  @rule = rul
end

def roll
  case rule
  when :each
    roll_all
  when :sum
    total
  when :worst
    take_worst
  when :best
    take_best
  end
end

def rule=(nu_rule)
  unless RULES.include?(nu_rule)
    str = %("#{nu_rule}")
    str << ' is not one of '
    str << RULES.map(&:inspect).join(', ')

    raise ArgumentError, str
  end

  @rule = nu_rule
end
