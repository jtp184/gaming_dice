def initialize(initarg = nil, discr = false)
  if initarg.is_a?(String)
    @dice = Dice.call(initarg).flatten
  elsif initarg.is_a?(Array)
    @dice = initarg
  elsif initarg.is_a?(Dice)
    @dice = Array(initarg)
  end

  @discrete = discr
end
